import java.util.Arrays;
import java.util.Stack;

import java.util.*;

class Node
{
   String data;
   Node next;
   
   public Node(String d, Node n)
   {
      this.data = d;
      next = n;
   }
}

public class SimpleLinkedList
{
   Node head = null;
   
   public SimpleLinkedList()
   {
      
   }
   public SimpleLinkedList(Node h)
   {
      head = h;
   }
   public void prepend(String d) // running time: O(1)
   
   {
      Node n = new Node(d,null);
      if (head == null)
      {
         head = n;
         
      }
      else
      {
         n.next = head;
         head = n; // update the head
         
      }
   }
   
   Node findTail()
   {
      Node x = head;
      if (x == null) return null;
         while (x.next != null)
         {
            x = x.next;
         }
         //at exit/ by the end of loop, x.next = null
         return x;
   }
   
   public void append(String d) // O(N)
   
   {
      
      Node tail = findTail();
      if (tail == null)
      {
         head = new Node(d,null);
      }
      else
      {
         tail.next = new Node(d,null);
      }
   }
   
   public int length() // O(N)
   
   {
      int leng = 0;
      if (head == null) return 0;
         Node x = head;
      // make sure the object/variable in front of "." is not null
      while (x.next != null)
      {
         x = x.next;
         leng++;
      }
      return leng + 1;
   }
   //       A->N->B       ->T          N
   // insert after no body
   
   void insertAfter(Node anchor, String data)
   {
      if (anchor == null)
      {
         //insert after no body
         prepend(data);
         return;
      }
      Node n = new Node(data, null);
      Node B = anchor.next;
      n.next = B;
      anchor.next = n;
      // anchor.next = new Node(data, anchor.next);
      
   }
   // p.next = n
   Node previous(Node n)
   {
      Node x = head;
      if (x == null) return null;
         while (x.next != null)
         {
            if (x.next == n)
               return x;
         }
         return null;
   }
   //      P N A 
   // insert before no body
   // tail->null
   void insertBefore(Node anchor, String data)
   {
      //insert before no body, want to become tail
      if (anchor == null)
      {
         append(data);
         return;
      }
      Node p = previous(anchor);
      if (p == null)
      {
         // anchor is the head
         prepend(data);
      }
      else
      {
         Node n = new Node(data, null);
         p.next = n;
         n.next = anchor;
         // p.next = new Node(data, anchor);
         
      }
      
   }
   
   void print()
   {
      Node x = head;
      while (x!=null)
      {
         System.out.print(x.data + "->");
         x = x.next;
      }
      System.out.println("null");
   }
   // p->x->y  =>  p->y->x
   public void swap(Node x)
   {
      Node y = x.next;
   }
   
   void bubblesort()
   {
      Node previous, current, nextone;
      if (head == null) return;
         int leng = length();
      for (int round = 1; round < leng; round++)
      {
         previous = null;
         current = head;
         nextone = current.next;
         while (nextone!=null)
         {
            if (nextone.data.compareTo(current.data) < 0)
            {
               current.next = nextone.next;
               nextone.next = current;
               if (previous!=null)
               {
                  //p->c->n->n.n -> p->n->c->n.n -> p->c->n->n.n
                  previous.next = nextone;
               }
               else // current -> nextone
                  
               {
                  head = nextone;
               }
               previous = nextone;
               nextone = current.next;
            }
            else
            {
               previous = current;
               current = nextone;
               nextone = nextone.next;
               
            }
            
         }
      }
   }
   // head->a->b->c
   Node removehead()
   {
      if (head == null) return null;
         Node x = head;
      head = head.next;
      x.next = null;
      return x;
   }
   Node removetail()
   {
      if (head == null) return null;
         Node x = head;
      Node p = null; // previous, old
      while (x.next != null)
      {
         p = x;
         x = x.next;
      }
      // by exit of loop, x is the desired tail
      if (p != null)
      {
         p.next = null;
      }
      else
      {
         head = null;
      }
      return x;
   }
   void sort()
   {
      if (head == null) return;
         int N = length();
      if (N == 1) return;
         String [] d = new String[N];
      Node x = head;
      int i = 0;
      while (x!=null)
      {
         d[i++] = x.data;
         x = x.next;
      }
      // by exit, i = N; x = null;
      Arrays.sort(d);
      while ( i > 0)
         x = new Node(d[--i],x);
      head = x;
      
   }
   Node removeAfter(Node anchor)
   {
      if (head == null) return null;
         if (anchor == null)
         {
            return removehead();
         }
         Node x = anchor.next;
         if (x == null)
         {
            return null;
         }
         anchor.next = x.next;
         x.next = null;
         return x;
   }
   
   //navigate through the list, for each node n, compare its data with key, when
   //matching, increment the count, finally return the count
   int count(String key) // better called it frequency
   
   {
      if (head == null) return 0;
         int counter = 0;
      Node n = head;
      while (n!=null)
      {
         if (n.data.equals(key))
            counter++;
         n = n.next;
      }
      return counter;
   }
   //navigate through the list, for each node n encountered, compare its data with key, when
   //matching, delete the node, then quit and return n
   Node removeFirst(String key) //
   
   {
      if (head == null)
         return null;
      Node n = head;
      Node p = null;
      while (n!=null)
      {
         if (n.data.equals(key))
         {
            if (p == null)
            {
               head = head.next;
            }
            else
            {
               p.next = n.next;
            }
            n.next = null;
            break;
         }
         else
         {
            p = n;
            n = n.next;
         }
      }
      // your codes go here. three node are invloved: previous, n and n's next
      return n;
   }
   //navigate through the list, for each node n, compare its data with key, when
   //matching, delete the node, keep going until the end
   void removeAll(String key) //
   
   {
      if (head == null) return;
         Node n = head;
      Node p = null;
      while (n!=null)
      {
         if (n.data.equals(key))
         {
            if (p == null)
            {
               head = head.next;
               n.next = null;
               n = head;
            }
            else
            {
               p.next = n.next;
               n.next = null;
               n = p.next;
            }
         }
         else
         {
            p = n;
            n = n.next;
         }
      }
      
      // your codes go here. three nodes are invloved: previous, n and n's next
      
   }
   
   void reversePrint()
   {
      Stack<String> s = new Stack();
      Node x = head;
      while (x!=null)
      {
         s.push(x.data);
         x = x.next;
      }
      while (!s.isEmpty())
      {
         String d = s.pop();
         System.out.println(d);
      }
   }

   {
      SimpleLinkedList simplelist = new SimpleLinkedList();

   }
