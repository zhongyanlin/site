   int compareTo(int i, int j)
   {
      numOfcomparisons++;
      return x[i].compareTo(x[j]);
   }
   
   void swap(int m, int n)
   {
      temp = x[m];
      x[m] = x[n];
      x[n] = temp;
      numOfcopies += 3;
   }
   
   void moveToy(int from, int to)
   {
      numOfcopies++;
      y[to] = x[from];
   }
   
   void moveTox(int from, int to)
   {
      numOfcopies++;
      x[to] = y[from];
   }
   
   void moveTo(int from, int to)
   {
      numOfcopies++;
      x[to] = x[from];
   }

      numOfcopies++;

   int maxima(int n)

         swap(m, n);

   {

      for (k = 0; k < n - 1; k++)

         int low = 0, high = k + 1, m = 0;

   {
      int midpoint = (highIndex + lowIndex) / 2;
      int lowIndex0 = lowIndex;
      int highIndex0 = highIndex;
      boolean done = false;
      while (!done)
      {
         while (midpoint != lowIndex)
         {
            int diff = compareTo(lowIndex, midpoint);
            if (diff < 0)
            {
               lowIndex++;
            }
            else
            {
               break;
            }
         }
         while (true)
         {
            int diff = compareTo(midpoint, highIndex);
            if (diff < 0)
            {
               highIndex--;
            }
            else
            {
               break;
            }
         }
         if (lowIndex >= highIndex)
         {
            done = true;
         }
         else
         {
            swap(lowIndex, highIndex);
            if (midpoint == lowIndex)
            {
               midpoint = highIndex;
            }
            else if (midpoint == highIndex)
            {
               midpoint = lowIndex;
            }
            lowIndex += 1;
            highIndex -= 1;
         }
      }
      return highIndex;
   }

      if (leftq.size() != 0 && rightq.size() != 0)

         i = leftq.poll();

         return new int[]
         {
            i, j
         };
      }
      if (leftq.size() != 0)
      {
         
         i = j = leftq.poll();
         while (leftq.isEmpty() == false)
         {
            j = leftq.poll();
         }
         return new int[]
         {
            i, j
         };
      }
      i = j = rightq.poll();
      while (rightq.isEmpty() == false)
      {
         j = rightq.poll();
      }
      return new int[]
      {
         j, i
      };
   }

   void merge(int low, int high)
   {
      int m = (low + high) / 2;
      int i = low;
      int j = m + 1;
      int k = low;
      for (; i <= m && j <= high;)
      {
         int diff = compareTo(i, j); // diff < 0 <=> x[i] < x[j]
         if (diff < 0)
         {
            moveToy(i, k);//y[k] = x[i]; 
            i++;
            k++;
         }
         else
         {
            moveToy(j, k);//y[k] = x[j]; 
            j++;
            k++;
         }
      }
      for (; i <= m;)
      {
         moveToy(i, k);//y[k] = x[i];
         i++;
         k++;
      }
      for (; j <= high;)
      {
         moveToy(j, k);//y[k] = x[j];
         j++;
         k++;
      }
      
      for (i = low; i <= high; i++)
      {
         moveTox(i, i);//x[i] = y[i];
         
      }
   }

   {
      int m = (low + high) / 2;
      int i = low;
      int j = m + 1;
      int k = low;
      int start2move = high + 1;
      for (; i <= m && j <= high;)
      {
         int diff = compareTo(i, j); // diff < 0 <=> x[i] < x[j]
         if (diff < 0)
         {
            if (start2move < high + 1)
            {
               moveToy(i, k);//y[k] = x[i]; 
               
            }
            i++;
            k++;
         }
         else
         {
            if (start2move == high + 1)// initial
               
            {
               start2move = k; //
               
            }
            moveToy(j, k);//y[k] = x[j]; 
            j++;
            k++;
         }
      }
      if (i <= m)// move remaining elements in the 1st sub array to the tail
         
      {
         for (i = high; i >= k; i--)
         {
            moveTo(m + i - high, i);//x[i] = x[m + i - high];

      }

      }
   }

   public void heapSort()
   {
      for (int i = N / 2 - 1; i >= 0; i--)
         heapify(N, i);
      for (int i = N - 1; i > 0; i--)
      {
         swap(i,0);
         heapify(i, 0);
      }
   }
   
   void heapify(int n, int i)
   {
      int largest = i;
      int l = 2 * i + 1; // left = 2*i + 1
      int r = 2 * i + 2; // right = 2*i + 2
      if (l < n )
      {
         int diff = compareTo(l, largest);//&& x[l] > arr[largest]
         if (diff > 0)
            largest = l;
      }
      if (r < n )
      {
         int diff = compareTo(r,largest);
         if (diff > 0)
            largest = r;
      }
      if (largest != i)
      {
         swap(i, largest);
         heapify(n, largest);
      }
   }

   int RadixGetMaxLength(int[] array, int arraySize)
   {
      int maxDigits = 0;
      for (int i = 0; i < arraySize; i++)
      {
         int digitCount = RadixGetLength(array[i]);
         if (digitCount > maxDigits)
         {
            maxDigits = digitCount;
         }
      }
      return maxDigits;
   }
   
   void RadixSort(int[] array)
   {
      int arraySize = array.length;
      ArrayList<Integer>[] buckets = new ArrayList[10];
      for (int i = 0; i < buckets.length; i++)
      {
         buckets[i] = new ArrayList();
      }
      int maxDigits = RadixGetMaxLength(array, arraySize);
      // Start with the least significant digit
      int pow10 = 1;
      for (int digitIndex = 0; digitIndex < maxDigits; digitIndex++)
      {
         for (int i = 0; i < arraySize; i++)
         {
            int bucketIndex = Math.abs(array[i] / pow10) % 10;
            buckets[bucketIndex].add(array[i]);
         }
         int arrayIndex = 0;
         for (int i = 0; i < 10; i++)
         {
            for (int j = 0; j < buckets[i].size(); j++)
            {
               array[arrayIndex++] = buckets[i].get(j);
            }
         }
         pow10 = 10 * pow10;
         for (int i = 0; i < 10; i++)
         {
            buckets[i].clear();
         }
      }
   }
   // Returns the length, in number of digits, of value
   
   int RadixGetLength(int value)
   {
      if (value == 0)
      {
         return 1;
      }
      int digits = 0;
      while (value != 0)
      {
         digits = digits + 1;
         value = value / 10;
      }
      return digits;
   }
