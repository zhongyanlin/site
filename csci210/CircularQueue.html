public class CircularQueue
{
   private String [] elements;
   private int front = -1, tail = -1;
   public CircularQueue(int cap)
   {
      elements = new String[cap];
      
   }
   public CircularQueue()
   {
      elements = new String[100];
   }
   public int length()
   {
      if (tail <= front)
         return front-tail+1;
      return
      elements.length - (tail - front - 1);
   }
   public void resize(int N) // not a frequent method
   
   {
      if (tail == -1)
      {
         elements = new String[N];
         return;
      }
      String [] x = new String[N];
      int i = tail;
      int j = 0;
      int N0 = elements.length;
      for(; j < N; i++,j++)
      {
         x[j] = elements[i % N0];
         if (i%N0 == front) break;
      }
      tail = 0;
      if (j == N)
         front = N-1;
      else
         front = j;
      elements = x;
   }
   public boolean enqueue(String x) // when q is full, enqueue fails
   
   {
      int N = elements.length;
      if (tail == -1)
      {
         tail = front = 0;
         elements[0] = x;
         return true;
      }
      else
      {
         int i = (tail-1 + N) % N; // intended
         if (i != front)
         {
            elements[i] = x;
            tail = i;
            return true;
         }
      }
      return false;
   }
   
   public String dequeue() // when q is empty, return null
   
   {
      int N = elements.length;
      if (front == -1)
         return null;
      String x = elements[front];
      if (front == tail)
      {
         front = tail = -1;
      }
      else
      {
         front = (front - 1 + N)%N;
      }
      return x;
   }
   
   public static void main(String [] args)
   {
      CircularQueue q = new CircularQueue(5);
      q.enqueue("A");
      q.enqueue("B");
      q.enqueue("C");
      q.enqueue("D");
      q.enqueue("E");
      System.out.println("q.length="+ q.length());
      q.enqueue("F");
      System.out.println("q.length="+ q.length());
      System.out.println("dequeue = "+ q.dequeue());
      System.out.println("q.length="+ q.length());
      q.resize(7);
      q.enqueue("C");
      q.enqueue("D");
      System.out.println("q.length="+ q.length());
   }
}
