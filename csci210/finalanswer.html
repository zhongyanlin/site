9. Node reverse(Node head)
{
if (head == null

head.next == null)
return head;
Node tail = head;
Stack<Node> stack = new Stack<>();
while (tail.next != null)
{
stack.add(tail);
tail = tail.next;
}
head = tail;
Node x, y=stack.pop();
while (stack.isEmpty() == false)
{
x = stack.pop();
y.next = x;
y = x;
}
x.next = null;
return head;
}
